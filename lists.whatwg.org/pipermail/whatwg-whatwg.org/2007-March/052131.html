>    <label>Species:
>      <input type="text" name="species[observation].name" value="" />
>    </label>
>    <label>Count:
>      <input type="number" min='1'  value=""
>             name="species[observation].number" />
>    </label>
>    <button type="remove" template="observation">Delete</button>

> </div>
