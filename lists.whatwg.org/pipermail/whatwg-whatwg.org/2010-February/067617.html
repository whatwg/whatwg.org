The worker script could be modified in step 3 as follows to make it clear  
that the script is in fact shared:

test.js
var i = 0;
onconnect = function(e) {
   i++;
   var port = e.ports[0];
   port.postMessage('hello, ' + i);
   port.onmessage = function(e) {
     port.postMessage('pong');
   }
}
