You could also add a listener to the image to check that it actually loads:

 

$(document).ready(function() {

      var image = $("#cat").get(0);

      image.onload = function(e){

          var cv = $("#img_container").get(0);      

          var ctx = cv.getContext('2d');

      

          ctx.drawImage(image, 0, 0);

            };

});
