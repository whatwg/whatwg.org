+  <head>

+  </head>
+  <body>
+    <p>Welcome to this example.</p>
+  </body>
+</html></pre>
+

+
+   <pre><!DOCTYPE HTML>
+<!-- where is this comment in the DOM? -->

+  </head>
+  <body>
+    <p>Welcome to this example.</p>
+  </body>
+</html></pre>
